ការឆ្លើយនឹងសំណួរដោយផ្អែកលើអំណះអំណាង

ដើម្បីឆ្លើយនឹងសំណួរ ឬពន្យល់គំនិតអ្វីមួយឱ្យកាន់តែច្បាស់លាស់ គេតែងលើកយកអំណ៖អំណាង ឬភស្តុតាងជាក់ស្តែងដែលធ្លាប់ជួបប្រទ៖ ឬដកស្រង់ចេញពីក្នុងរឿងណាមួយមកបញ្ជាក់ ។

ចូរអានរឿងខាងក្រោម និងឆ្លើយនឹងសំណួរដោយស្រង់អំណះអំណាងពីរឿងមកបង្ហាញ ។

រឿង ចាបស្រុកនិងចាបព្រៃ

កាលពីព្រេងនាយ មានចាបស្រុក និងចាបព្រៃដើររកស៊ីដោយពួកជារៀងរាល់ថ្ងៃ ។ខាងចាបស្រុកមានមេប្រដៅថា បើរកស៊ីកុំជជែកឈ្លោះគ្នាឱ្យឮមាត់ ឱ្យប្រយ័ត្នខ្លួន បើព្រានវាយលប់ត្រូវ ឱ្យបបួលគ្នាខំហើរទូលលប់ឱ្យរបើក ដើម្បីឱ្យរស់រួចជីវិត ។ ពួកចាបស្រុកធ្វើតាមដំបូន្មាន បានរួចជីវិតតរៀងៗមក ។ មានកាលថ្ងៃមួយ ចាបព្រៃនិងចាបស្រុកហើរចេញទៅរកស៊ី ក៏ជួបបទ៖ហ្វូងគ្នានៅលើដើមត្រែងមួយគុម្ព ។ ពួកចាបពទំនៅខាងក្រោម ពួកចាបស្រុកទំនៅខាងលើក៏បាក់មែកត្រែងពីលើមកត្រូវក្បាលចាបព្រៃ ។ចាបព្រៃនិយាយប្រកាន់ថា ពួកចាបស្រុកកាច់មែកត្រែងទម្លាក់មកឱ្យត្រូវកមកឱ្យត្រូវក្បាលពួកខ្លួន ។ចាបស្រុកបានប្រាប់ថា ពួកខ្លួនមិនបានកាច់មែកត្រែងឱ្យបាក់ទេ គឺមកត្រែងវាបាក់ធ្លាក់ខ្លួនឯងទ ។ ពួកចាបព្រៃមិនពិចារណាក៏បបួលគ្នាបង្កើតជម្លោះនឹងពួកចាបស្រុក រហូតឮមាត់គឹកកងរំពងឡើង ។ ព្រានឮមាត់ចាប ក៏យកលប់ទៅវាយត្រូវទាំងអស់គ្នា ។ ចាបទាំងពីរពួកនៅតែឈ្លោះប្រកែកគ្នា ទាល់តែព្រានចាប់បានពិឃាតក្ស័យអស់ទៅ ។

(ដកស្រង់និងសម្រួលពីប្រជុំរឿងព្រេងខ្មែរភាគទី ២)

ចម្លើយ ៖ បានជាព្រានសម្លាប់ពួកចាបទាំងពីរពួកបាន ពីព្រោះពួកចាបទាំងពីរពួកមិនមូលកម្លាំងគ្នារកវិធីដោះខ្លួនចេញពីលប់ទេ ដោយរវល់តែឈ្លោះប្រកែកគ្នា ។ ជាពិសស ចាបស្រុកភ្លេចដំបូន្មានីមេបា ។
